**Analog:** itself (same function/file), reusing the mechanism Phase 22.1 already built at

**Current code, exact excerpt (867-948)** — this IS both the analog and the edit target, so quote
it verbatim rather than a separate file:

```python
    def finish(self) -> None:
        """
        Finish the build process by compiling Typst files to PDF.
        ...
        Requirement 9.2: Execute Typst compilation within Python
        Requirement 9.4: Generate PDF from Typst markup
        """
        # First, call parent finish() to copy image files
        super().finish()

        # Get master documents from typst_documents config
        typst_documents = getattr(self.config, "typst_documents", [])

        if not typst_documents:
            logger.warning(
                "No documents defined in typst_documents. Nothing to compile."
            )
            return

        logger.info(f"Compiling {len(typst_documents)} master document(s) to PDF...")

        failures: List[Tuple[str, str]] = []

        for doc_tuple in typst_documents:
            if not doc_tuple:                                    # <- 912-916, D-05/D-07 edit site
                logger.warning(
                    f"Skipping malformed typst_documents entry: {doc_tuple!r}"
                )
                continue
            docname = doc_tuple[0]
            stem = self._resolve_output_stem(docname)
            relative_path = self._directory_preserving_relpath(docname, stem)
            typ_file = path.join(self.outdir, relative_path + ".typ")

            if not path.exists(typ_file):                        # <- 922-924, D-01/D-04 edit site
                logger.warning(f"Master document not found: {typ_file}")
                continue

            try:
                pdf_bytes = compile_typst_file_to_pdf(typ_file, root_dir=self.outdir)
                pdf_file = path.join(self.outdir, relative_path + ".pdf")
                with open(pdf_file, "wb") as f:
                    f.write(pdf_bytes)
                logger.info(f"Generated PDF: {pdf_file}")

            except Exception as e:                                # <- 939-941, EXISTING failures pattern
                logger.error(f"Failed to compile {typ_file}: {e}")
                failures.append((docname, str(e)))

        if failures:                                              # <- 943-948, D-02 message-broadening site
            summary = "; ".join(f"{docname}: {err}" for docname, err in failures)
            raise ExtensionError(
                f"typstpdf: {len(failures)} master document(s) failed to compile "
                f"to PDF: {summary}"
            )
```

**Pattern to copy — the existing `failures.append` shape at 939-941** is the exact template for
both new call sites. It is a 2-tuple `(docname_or_repr, message)`; no new data structure needed.
Both new guards should mirror this shape: `logger.warning(msg)` kept as-is, then
`failures.append((identifier, msg))`, then `continue` — a pure 1-line insertion at each site,
matching D-05/D-07/D-01/D-04's stated minimal-diff intent.

**D-04 discrimination source (read-only reference, do not modify):** `self.env.found_docs` is a
`set[str]` populated by Sphinx core before `finish()` ever runs (confirmed live in RESEARCH.md's
Code Examples section: `app.env.found_docs` populated at `SphinxTestApp` construction). Use
`docname not in self.env.found_docs` as the discriminator; on the "other cause" branch, the
fallback message text `f"Master document not found: {typ_file}"` **must stay byte-identical** —
see Shared Patterns below for why.

**D-02 aggregate message:** the ONLY string literal edit needed is dropping "to compile" from
`"...failed to compile to PDF: ..."` → `"...failed: ..."` (or similarly broadened wording); keep
the `typstpdf:` prefix (no underscore — confirmed live, contra CONTEXT.md's `typst_pdf:` quoting
error) and the `"; ".join(...)` summary format unchanged, since `failures` stays a
`List[Tuple[str, str]]` regardless of which branch appended to it.

```python
import subprocess
import sys
from pathlib import Path

import pytest

try:
    import typst
    TYPST_AVAILABLE = True
except ImportError:
    TYPST_AVAILABLE = False

FIXTURES_DIR = Path(__file__).parent / "fixtures"
GATE_FIXTURE_DIR = FIXTURES_DIR / "missing_and_malformed_master_gate"


def _run_sphinx_build(
    source_dir: Path, build_dir: Path, builder: str
) -> subprocess.CompletedProcess:
    """
    Run `sphinx-build -b <builder>` as a subprocess... Invoked as
    `sys.executable -m sphinx` (never `uv run sphinx-build`) so the exact
    interpreter/venv running this test is reused, sidestepping the
    documented NixOS-sandbox PATH-shadowing hazard.
    """
    return subprocess.run(
        [sys.executable, "-m", "sphinx", "-b", builder, str(source_dir), str(build_dir)],
        capture_output=True,
        text=True,
    )
```

**Inverted assertion (new in this suite; NO existing sibling to copy verbatim, described in
RESEARCH.md Pattern 1):**
```python
@pytest.mark.skipif(not TYPST_AVAILABLE, reason="typst-py is required for the ... gate")
class TestMissingAndMalformedMasterGate:
    def test_mixed_bad_entries_fail_build_but_good_master_still_compiles(
        self, tmp_path
    ):
        build_dir = tmp_path / "_build"
        result = _run_sphinx_build(GATE_FIXTURE_DIR, build_dir, "typstpdf")

        # THIS IS THE FIRST "MUST-FAIL" ASSERTION IN THE SUITE -- every
        # sibling GATE-01 module asserts == 0. Do not hardcode returncode
        # == 2 (that is Sphinx CLI's own convention); assert != 0.
        assert result.returncode != 0, (
            f"Expected a non-zero exit for a project with an unknown-docname "
            f"master and a malformed typst_documents entry, got 0:\n"
            f"stdout: {result.stdout}\nstderr: {result.stderr}"
        )
        # Assert on typsphinx's OWN ExtensionError substring -- never
        # typst-py's diagnostic text (D-10's scope is typst-py specifically;
        # this is unaffected).
        assert "master document(s) failed" in result.stderr

        # D-02 attempt-all-then-raise survives mixed good/bad entries: the
        # good master still gets its PDF even though the build overall fails.
        assert (build_dir / "index.pdf").exists()
```
